python動画で勉強用に紹介されていたコード

               
                import tkinter


                class Application(tkinter.Frame):
                 def __init__(self, root=None):
                     super().__init__(root, width=380, height=280,
                         borderwidth=4, relief='groove')
                    self.root = root
                    self.pack()
                    self.pack_propagate(0)
                    self.create_widgets()

                　def create_widgets(self):
                     quit_btn = tkinter.Button(self)
                     quit_btn['text'] = '閉じる'
                     quit_btn['command'] = self.root.destroy
                     quit_btn.pack(side='bottom')
                     # テキストボックス
                     self.text_box = tkinter.Entry(self)
                     self.text_box['width'] = 10
                     self.text_box.pack()
                     # 実行
                     submit_btn = tkinter.Button(self)
                     submit_btn['text'] = '実行'
                     submit_btn['command'] = \
                     self.input_handler
                     submit_btn.pack()
                     # メッセージ
                     self.message = tkinter.Message(self)
                     self.message.pack()
        
                     def input_handler(self):
                        text = self.text_box.get()
                        self.message['text'] = text + '!!'

                root = tkinter.Tk()
                root.title('サンプルアプリ')
                root.geometry('400x300')
                app = Application(root=root)
                app.mainloop()

arXivから論文の概要を取得するコード

chatGPTに教えてもらったコードです。queryの行ではAIに関する2023年の論文を取得するように指定しています。少しいじくれば条件を変えて取得することもできます。

                
                import feedparser
                import requests

                # arXiv APIのURLと検索クエリを設定する
                arxiv_url = "http://export.arxiv.org/api/query"
                query = "search_query=cat:cs.AI&sortBy=submittedDate&sortOrder=descending&start=0&max_results=10&date-range=2023-01-01%3A2023-12-31"
                # 論文を取得する
                response = requests.get(arxiv_url, params=query)
                # 取得した論文をパースして情報を抽出する
                feed = feedparser.parse(response.text)
                for entry in feed.entries:
                # 論文情報を取得する
                title = entry.title
                authors = ", ".join(author.name for author in entry.authors)
                abstract = entry.summary
                published = entry.published

                # 結果を出力する
                print(f"Title: {title}")
                print(f"Authors: {authors}")
                print(f"Abstract: {abstract}")
                print(f"Published: {published}")
                print()
